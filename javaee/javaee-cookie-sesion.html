AddCookies.java

import java.io.IOException;
import java.io.PrintWriter;
import java.net.URLEncoder;

import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.Cookie;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

@WebServlet("/servlet/AddCookies")
public class AddCookies extends HttpServlet {
    private static final long serialVersionUID = 1L;

    /**
     * @see HttpServlet#HttpServlet()
     */
    public AddCookies() {
        super();
    }

    /**
     * @see HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
     */
    public void doGet(HttpServletRequest request, HttpServletResponse response)
                    throws ServletException, IOException {
        // СИ║тљЇтГЌтњїтДЊТ░ЈтѕЏт╗║ Cookie
        Cookie name = new Cookie("name", URLEncoder.encode(request.getParameter("name"), "UTF-8")); // СИГТќЄУйгуаЂ
        Cookie url = new Cookie("url", request.getParameter("url"));

        // СИ║СИцСИф Cookie У«Йуй«У┐ЄТюЪТЌЦТюЪСИ║ 24 т░ЈТЌХтљј
        name.setMaxAge(60 * 60 * 24);
        url.setMaxAge(60 * 60 * 24);

        // тюетЊЇт║ћтц┤СИГТи╗тіаСИцСИф Cookie
        response.addCookie(name);
        response.addCookie(url);

        // У«Йуй«тЊЇт║ћтєЁт«╣у▒╗тъІ
        response.setContentType("text/html;charset=UTF-8");

        PrintWriter out = response.getWriter();
        String title = "У«Йуй« Cookie т«ъСЙІ";
        String docType = "<!DOCTYPE html>\n";
        out.println(docType + "<html>\n" + "<head><title>" + title + "</title></head>\n"
                        + "<body bgcolor=\"#f0f0f0\">\n" + "<h1 align=\"center\">" + title
                        + "</h1>\n" + "<ul>\n" + "  <li><b>уФЎуѓ╣тљЇ№╝џ</b>№╝џ" + request.getParameter("name")
                        + "\n</li>" + "  <li><b>уФЎуѓ╣ URL№╝џ</b>№╝џ" + request.getParameter("url")
                        + "\n</li>" + "</ul>\n" + "</body></html>");
    }

    /**
     * @see HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
     */
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
                    throws ServletException, IOException {
        doGet(request, response);
    }

}

# Cookie vs Session

# тГўтЈќТќ╣т╝Ј

Cookie тЈфУЃйС┐ЮтГўASCIIтГЌугдСИ▓№╝їтдѓТъюжюђУдЂтГўтЈќ Unicode тГЌугдТѕќС║їУ┐ЏтѕХТЋ░ТЇ«№╝їжюђУдЂУ┐ЏУАїUTF-8сђЂGBKТѕќBASE64уГЅТќ╣т╝Јуџёу╝ќуаЂсђѓ

Session тЈ»С╗ЦтГўтЈќС╗╗СйЋу▒╗тъІуџёТЋ░ТЇ«№╝їућџУЄ│Тў»С╗╗СйЋ Java у▒╗сђѓтЈ»С╗Цт░є Session уюІТѕљТў»СИђСИф Java т«╣тЎеу▒╗сђѓ

# жџљуДЂт«ЅтЁе

Cookie тГўС║јт«бТѕиуФ»ТхЈУДѕтЎе№╝їСИђС║Џт«бТѕиуФ»уџёуеІт║ЈтЈ»УЃйС╝џуфЦТјбсђЂтцЇтѕХТѕќС┐«Тћ╣ Cookie тєЁт«╣сђѓ

Session тГўС║јТюЇтіАтЎе№╝їт»╣т«бТѕиуФ»Тў»жђЈТўјуџё№╝їСИЇтГўтюеТЋЈТёЪС┐АТЂ»Т│ёжю▓уџётЇ▒жЎЕсђѓ

# ТюЅТЋѕТюЪ

Сй┐уће Cookie тЈ»С╗ЦС┐ЮУ»ЂжЋ┐ТЌХжЌ┤уЎ╗тйЋТюЅТЋѕ№╝їтЈфУдЂУ«Йуй« Cookie уџёmaxAgeт▒ъТђДСИ║СИђСИфтЙѕтцДуџёТЋ░тГЌсђѓ

Уђї Session УЎйуёХуљєУ«║СИіС╣ЪтЈ»С╗ЦжђџУ┐ЄУ«Йуй«тЙѕтцДуџёТЋ░тђ╝ТЮЦС┐ЮТїЂжЋ┐ТЌХжЌ┤уЎ╗тйЋТюЅТЋѕ№╝їСйєТў»№╝їућ▒С║ј Session СЙЮУхќС║јтљЇСИ║JESSIONIDуџё Cookie№╝їУђї Cookie JESSIONIDуџёmaxAgeж╗ўУ«цСИ║-1№╝їтЈфУдЂтЁ│жЌГС║єТхЈУДѕтЎеУ»Ц Session т░▒С╝џтц▒ТЋѕ№╝їтЏаТГц№╝їSession СИЇУЃйт«ъуј░С┐АТЂ»Т░ИС╣ЁТюЅТЋѕуџёТЋѕТъюсђѓСй┐уће URL тю░тЮђжЄЇтєЎС╣ЪСИЇУЃйт«ъуј░сђѓ

# ТюЇтіАтЎеуџёт╝ђжћђ

ућ▒С║ј Session Тў»С┐ЮтГўтюеТюЇтіАтЎеуџё№╝їТ»ЈСИфућеТѕижЃйС╝џС║ДућЪСИђСИф Session№╝їтдѓТъют╣ХтЈЉУ«┐жЌ«уџёућеТѕижЮътИИтцџ№╝їС╝џС║ДућЪтЙѕтцџуџё Session№╝їТХѕУђЌтцДжЄЈуџётєЁтГўсђѓ

Уђї Cookie ућ▒С║јС┐ЮтГўтюет«бТѕиуФ»ТхЈУДѕтЎеСИі№╝їТЅђС╗ЦСИЇтЇаућеТюЇтіАтЎеУхёТ║љсђѓ

# ТхЈУДѕтЎеуџёТћ»ТїЂ

Cookie жюђУдЂТхЈУДѕтЎеТћ»ТїЂТЅЇУЃйСй┐ућесђѓ

тдѓТъюТхЈУДѕтЎеСИЇТћ»ТїЂ Cookie№╝їжюђУдЂСй┐уће Session С╗ЦтЈі URL тю░тЮђжЄЇтєЎсђѓ

жюђУдЂТ│еТёЈуџёС║ІТЅђТюЅуџёућетѕ░ Session уеІт║Јуџё URL жЃйУдЂСй┐ућеresponse.encodeURL(StringURL) Тѕќresponse.encodeRediretURL(String URL)У┐ЏУАї URL тю░тЮђжЄЇтєЎ№╝їтљдтѕЎт»╝УЄ┤ Session С╝џУ»ЮУиЪУИфтц▒ТЋѕсђѓ
